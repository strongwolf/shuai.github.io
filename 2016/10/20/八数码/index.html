八数码问题

Veröffentlicht am 2016-10-20

###1问题简介

八数码问题也称为九宫问题。在3×3的棋盘，摆有八个棋子，每个棋子上标有1至8的某一数字，不同棋子上标的数字不相同。棋盘上还有一个空格，与空格相邻的棋子可以移到空格中。要求解决的问题是：给出一个初始状态和一个目标状态，找出一种从初始转变成目标状态的移动棋子步数最少的移动步骤。

所谓问题的一个状态就是棋子在棋盘上的一种摆法。棋子移动后，状态就会发生改变。解八数码问题实际上就是找出从初始状态到达目标状态所经过的一系列中间过渡状态。

###2算法分析

(1)状态表示

每个棋子的位置用数字表示，空格用零表示，每个状态中棋子的位置存放在一维数组P[N]中。

(2)节点

问题的状态用节点表示，每个节点除了记录每个棋子的位置、空格的位置外，还用last记录了搜索路径中上一个节点。

(3)链表

将每次新扩展的节点按照代价函数的值按从小到大的顺序插入到链表中，并用head和tail记录未扩展节点在链表中的位置。每次新扩展的节点在加入链表前都要判断是否达到目标状态。

(4)搜索策略

每次选取为扩展节点中代价函数值最小的节点扩展，即head指向的节点，是一种全局最优策略。每次扩展前判断是否到达目标状态。F(n)=g(n)+h(n),g为节点在搜索树中的深度，h为棋子“不在位”的距离和。

#include<iostream>
    #include<stdlib.h>
    #include<stdio.h>
    #include<cmath>
    #include<time.h>
    using namespace std;
    #define N 9
    //结构体存放每个节点的状态
    typedef struct Node
    {  int p[N],r[N];
       int last,space;
       int f,g,h;
    }Node;
    //用链表存放新扩展的节点
    typedef struct List
    {  Node T;
       List *next;
    }List;
    int h[4]={1,3,-1,-3};
    int h1[9]={0,0,0,1,1,1,2,2,2};
    int h2[9]={0,1,2,0,1,2,0,1,2};
    //head和tail标记为扩展节点的开始和末尾
    int head,tail,length;
    Node d,b;
    void printn(Node t)
    {   int i=0;
        cout<<"\t┏━┳━┳━┓"<<endl;
        for( i=0;i<2;i++)
          { cout<<"\t┃";
            for(int j=0;j<3;j++)
               cout<<t.p[3*i+j]<<" ┃";
            cout<<endl<<"\t┣━╋━╋━┫"<<endl;
          }
        cout<<"\t┃";
        for(int j=0;j<3;j++)
           cout<<t.p[3*i+j]<<" ┃";
        cout<<endl<<"\t┗━┻━┻━┛"<<endl;
    }
    //根据节点位置获取节点信息
    Node getdata(List*l,int k)
    {    List* t=l;
        for(int i=0;i<k;i++)
            {
                t=t->next;
            }
         return t->T;
    }
    //由最后一个节点输出路径
    void printl(List* l,Node t)
    {   if(t.last==-1)
           { printn(t);return;}
        else 
        {  Node n=getdata(l,t.last);
           printl(l,n);
           cout<<endl<<"              ↓"<<endl<<"              ↓"<<endl<<endl;
           printn(t);
        }
    }
    //判断能否扩展，i为扩展方向
    int ifextend(Node t,int i)
    {   int s=t.space;
        if((i==0&&s%3==2)||(i==1&&s>5)||(i==2&&s%3==0)||(i==3&&s<3))
            return 0;
        return 1;
    }
    //扩展新的节点
    Node extend(Node t,int k)
    {   int s=t.space;
        int temp=s+h[k];
        t.p[s]=t.p[temp];
        t.p[temp]=0;
        t.space=temp;
        for(int i=0;i<9;i++)
            t.r[t.p[i]]=i;
        t.g++;
        int sum=0;
        for(int i=0;i<9;i++)
            { int x1=h1[t.r[i]];
              int x2=h1[d.r[i]];
              int y1=h2[t.r[i]];
              int y2=h2[d.r[i]];
              sum+=abs(x1-x2)+abs(y1-y2);
            }
        t.h=sum;
        t.f=t.g+sum;
        return t;
    }
    //判断两个节点是否相同
    int cmpare(Node t,Node n)
    {  for(int i=0;i<9;i++)
          if(t.r[i]!=n.r[i])
          {
              return 0;
          }
        return 1;
    }
    //从链表中找出与新扩展节点相同的节点
    int repeat(List* l,Node t)
    {   for(int i=0;i<length;i++)
         {  if(l!=NULL&&cmpare(l->T,t))
              return i;
            l=l->next;
         }
        return -1;
    }
    //向链表中顺序插入新节点
    void add(List*l,Node n)
    {
        int i;
        for(i=head+1;i<length;i++)
            { Node t=getdata(l,i);
              if(t.f>=n.f)
                break;
            }    
        for(int j=1;j<=i-1;j++)
              l=l->next;
        List* k=(List*)malloc(sizeof(List));
        k->T=n;
        k->next=l->next;
        l->next=k;
        length++;
    }
    //搜索
    void seach()
    {   List*L=(List*)malloc(sizeof(List));
        L->T=b;
        L->next=NULL;
        length=1;head=0;tail=0;
        while(head<=tail)
        {   Node R;
            R=getdata(L,head);
            for(int i=0;i<4;i++)
            {   if(cmpare(R,d))
                   {
                    printl(L,R);flag=1;return;
                   }
                if(ifextend(R,i)==1)
                {   Node Q;
                    Q=extend(R,i);
                    int m=repeat(L,Q);
                    if(m>=0)
                        continue;
                    Q.last=head;
                    add(L,Q);
                   tail++;  
                }
            }
            head++;
        }

    }

            }
